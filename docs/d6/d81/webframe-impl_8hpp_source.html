45 convert_path_to_regex(std::string str, webframe& app)
46 {
47 static const std::string regexAnyChar = "A-Za-z_%0-9.-";
48 std::vector<std::string> v;
49 std::string format = "^";
50 for (size_t i = 0; i < str.size(); i++)
51 {
52 if (str[i] == '{')
53 {
54 std::string var_type;
55 for (i++; i < str.size() and str[i] != '}'; i++)
56 {
57 if (str[i] == ':')
58 break;
59 var_type += str[i];
60 }
61 if (var_type == "")
62 var_type = "string";
63 if (str[i] == '}')
64 {
65 i -= var_type.size() + 1;
66 var_type = "string";
67 }
68 v.push_back(var_type);
69 format += "(";
70 std::string curr_val_regex;
71 for (i++; i < str.size() and str[i] != '}'; i++)
72 {
73 if (str[i] == ':')
74 break;
75 curr_val_regex += str[i];
76 }
77 if (curr_val_regex == "string" or curr_val_regex == "text")
78 curr_val_regex = "[" + regexAnyChar + "]+";
79 if (curr_val_regex == "char" or curr_val_regex == "symbol")
80 curr_val_regex = "[" + regexAnyChar + "]";
81 if (curr_val_regex == "digit")
82 curr_val_regex = "[0-9]";
83 if (curr_val_regex == "number")
84 curr_val_regex = "[1-9][0-9]*";
85 if (curr_val_regex == "path")
86 curr_val_regex = "[" + regexAnyChar + "\\/]+";
87 format += curr_val_regex;
88 format += ")";
89 }
90 else if (str[i] == '/') {
91 format += "\\/";
92 }
93 else
94 {
95 format += str[i];
96 }
97 }
98 format += "$";
99 app.logger << format << '\n';
100 return {v, std::regex(format)};
101 }

275 std::regex pieces_regex(s.first.second);
276 if (std::regex_match(path, pieces_match, pieces_regex))
277 {
278 std::ssub_match sub_match = pieces_match[0];

464 std::shared_future<void> run(const char* PORT, const unsigned int cores, std::promise<void>* started = nullptr, bool limited = false, unsigned int requests = -1)
465 {
466 std::shared_ptr<std::promise<void>> running = std::make_shared<std::promise<void>>();
467
468 std::thread([&, this](std::shared_ptr<std::promise<void>> running) {
469 #ifdef _WIN32
470 //----------------------
471 // Initialize Winsock.
472 WSADATA wsaData;
473 this->logger << "Startup called\n";
474 int iResult = WSAStartup(MAKEWORD(2, 2), &wsaData);
475 this->logger << "Startup finished " << iResult << "\n";
476 if (iResult != NO_ERROR) {
477 this->logger << "WSAStartup failed with error: " << iResult << "\n";
478 running->set_value();
479 return;
480 }
481 #endif
482 this->logger << "Startup called\n";
483
484 const unsigned int threads = std::min(cores, limited ? requests : cores);
485 std::shared_ptr<thread_pool> threads_ptr = std::make_shared<thread_pool>(threads);
486
487 this->logger << "Thread pool generated\n";
488
489 int status;
490 struct addrinfo hints, *res;
491 int listener;
492
493 // Before using hint you have to make sure that the data structure is empty
494 memset(&hints, 0, sizeof hints);
495 // Set the attribute for hint
496 hints.ai_family = AF_INET; // We don't care V4 AF_INET or 6 AF_INET6
497 hints.ai_socktype = SOCK_STREAM; // TCP Socket SOCK_DGRAM
498 hints.ai_flags = AI_PASSIVE;
499 hints.ai_protocol = 0; /* Any protocol */
500 hints.ai_canonname = NULL;
501 hints.ai_addr = NULL;
502 hints.ai_next = NULL;
503
504 // Fill the res data structure and make sure that the results make sense.
505 status = getaddrinfo(NULL, PORT, &hints, &res);
506 if (status != 0)
507 {
508 this->logger << "getaddrinfo error: " << gai_strerror(status) << "\n";
509 running->set_value();
510 return;
511 }
512
513 // Create Socket and check if error occured afterwards
514 listener = socket(res->ai_family, res->ai_socktype, res->ai_protocol);
515 if (listener == -1)
516 {
517 this->logger << "socket error: " << gai_strerror(status) << "\n";
518 running->set_value();
519 return;
520 }
521
522 // Bind the socket to the address of my local machine and port number
523 status = bind(listener, res->ai_addr, res->ai_addrlen);
524 if (status < 0)
525 {
526 this->logger << "bind error: " << gai_strerror(status) << "\n";
527 running->set_value();
528 return;
529 }
530
531 status = listen(listener, 10);
532 if (status < 0)
533 {
534 this->logger << "listen error: " << gai_strerror(status) << "\n";
535 running->set_value();
536 return;
537 }
538
539 status = nonblock_config(listener);
540 if (status < 0)
541 {
542 this->logger << "nonblocking config error: " << gai_strerror(status) << "\n";
543 running->set_value();
544 return;
545 }
546
547 // Free the res linked list after we are done with it
548 freeaddrinfo(res);
549
550 this->logger << "Listener setup " << listener << "\n";
551
552 started->set_value();
553
554 while (!limited || requests != 0) {
555 const std::optional<size_t> thread = threads_ptr->get_free_thread();
556 if(!thread)
557 continue;
558
559 // Accept a new connection and return back the socket desciptor
560 threads_ptr->get(thread.value())->requestor = std::shared_ptr<int>(new int(ACCEPT(listener, NULL, NULL)));
561 if (*threads_ptr->get(thread.value())->requestor == -1)
562 {
563 continue;
564 }
565
566 this->logger << "Requestor " << *threads_ptr->get(thread.value())->requestor << " is getting handled\n";
567
568 struct timeval selTimeout;
569 selTimeout.tv_sec = 2;
570 selTimeout.tv_usec = 0;
571 fd_set readSet;
572 FD_ZERO(&readSet);
573 FD_SET(*threads_ptr->get(thread.value())->requestor + 1, &readSet);
574 FD_SET(*threads_ptr->get(thread.value())->requestor, &readSet);
575
576 this->logger << "Requestor " << *threads_ptr->get(thread.value())->requestor << " is checked if valid\n";
577
578 status = SELECT(*threads_ptr->get(thread.value())->requestor + 1, &readSet, nullptr, nullptr, &selTimeout);
579 this->logger << "SELECT status is " << status << "\n";
580 if (status <= 0)
581 continue;
582
583 this->logger << "Requestor " << *threads_ptr->get(thread.value())->requestor << " is still valid\n";
584
585 if (getsockname(*threads_ptr->get(thread.value())->requestor, nullptr, nullptr) < 0 && errno == ENOTSOCK) continue;
586
587 this->logger << "Requestor " << *threads_ptr->get(thread.value())->requestor << " is still valid\n";
588
589 this->logger << thread.value() << " thread will handle client " << *threads_ptr->get(thread.value())->requestor << "\n";
590
591 threads_ptr->get(thread.value())->detach(std::make_shared<std::function<void(int)>>([this, &limited, &running, &requests](int socket) -> void {
592 this->handler (socket, [&limited, &running, &requests]() {
593 if (!limited) return;
594 requests--;
595 if (requests == 0) {
596 running->set_value();
597 }
598 });
599 }), *threads_ptr->get(thread.value())->requestor);
600 }
601 #ifdef _WIN32
602 WSACleanup();
603 #endif
604 }, running).detach();
605 return running->get_future().share();
606 }
